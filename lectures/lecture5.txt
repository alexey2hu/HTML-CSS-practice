Что есть:
	1. Теги: Изображения
	2. Атрибуты (Изображения)

Изображения - теги:
	<!-- Image -->
	<img src="Ссылка" alt="Текстовое описание">

	<!-- Контейнер для 1 <img> и нескольких  <source> -->
	<!-- Нужен для адаптивной или кроссбраузерной верстки -->
	<picture>
		<source srcset="link3" media="(max-width: 500px)" />
		<source srcset="link2" media="(max-width: 1000px)" />
		<img src="link" alt="Текстовое описание" />
	</picture>

	<!-- Контейнер для <img> и <figcaption> -->
	<!-- Нужен для семантической верстки 
	(связывает изображение с его описанием) -->
	<figure>
		<img src="link" alt="Текстовое описание" />
		<figcaption>Описание картинки</figcaption>
	</figure>
