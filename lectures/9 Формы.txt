Что есть:
	1. Атрибуты: Формы
	2. Теги: Формы

Атрибуты (Формы):
	<!-- Путь к скрипту js обработки формы -->
		<form> action="link"
	<!-- BackEnd методы обращения -->
		<form> method="dialog/get/post"
	<!-- Разрешение авто-подстановки данных -->
		<form> autocomplete="off/on"
	
	<!-- Имя элемента (Индивидуальное) -->
	<!-- Используется для отправки на сервер -->
		<form>, <select> name="name"

	<!-- Значение для отправки на сервер при выборе -->
	<!-- Иначе используется вложенный текст -->
		<option> value="value"

	<!-- предположительное кол-во колонок и строк -->
	<!-- Задает изначальный размер поля ввода -->
		<textarea> cols="num" rows="num"

	<!-- Ссылка на id элемента для семантики -->
	<!-- Можно просто пометить элемент внутрь -->
		<label> for="#id"

Теги (Формы):
	Все: (<form></form>, <input>, <select></select>,
		<option></option>, <textarea></textarea>, <label></label>)

	<form>
		<!-- Надпись - подсказка по вводу -->
		<label>
			Подсказка:
			<!-- Поле для ввода данных -->
			<input type="text">
		</label>
		<!-- Поле для выбора данных -->
		<select name="selectName">
			<!-- Данные дял выбора -->
			<option value="default">-- По умолчанию --</option>
			<option value="1">-- Пункт 1 --</option>
			<option value="three">-- Пункт 2 --</option>
		</select>
		<!-- Поле для ввода предположительно БОЛЬШИХ данных -->
		<!-- Можно увеличивать или уменьшать на сайта -->
		<textarea name="areaName" cols="num" rows="num"></textarea>

		<!-- Кнопка отправки -->
		<button type='submit'>Отправить данные</button>
	</form>

Теги (Формы - доп.):
	<!-- Добавляет границу для данных внутри (Объединение тематически) -->
	<fieldset>
		<legend>Заголовок сверху</legend>
		<input type="text">
		<input type="text">
	</fieldset>
